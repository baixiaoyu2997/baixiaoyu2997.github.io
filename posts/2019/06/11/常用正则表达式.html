# 手机号

/* 是否是手机号码
 *  暂时不验证第二位是否是 3|4|5|7|8，不排除以后运营商增加其他网段
 *  如果必须验证第二位，可以替换成： reg = /^1(3|4|5|7|8)\d{9}$/
 * */
function isMobilePhone(str) {
  str = str + '' || ''
  const reg = /^1[0-9]{10}$/
  return reg.test(str.trim())
}

# 电话号

function isTel(str) {
  return /^0\d{2,3}-?\d{7,8}$/.test(str.trim())
}
