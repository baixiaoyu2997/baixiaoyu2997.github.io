# admin端

技术栈：基于umi v2.1.2 + create-umi v0.4.0 + antd pro v2.1.0开发

权限控制流程：
一、 进入路由

// config/router.config.js
 {
    path: '/',
    component: '../layouts/BasicLayout',
    Routes: ['src/pages/Authorized'], // 通过在BasicLayout布局组件外包裹Authorized权限校验组件进行权限判断
    authority: ['*LOGIN*'],
    routes: [
      // ....
    ],
  },

二、 权限判断

// src/pages/Authorized.js
import React from 'react';
import RenderAuthorized from '@/components/Authorized'; 
// 通过utils/authority.js设置和获取当前用户权限，原理是保存在`window._authority`对象上，例如：`['*LOGIN*']`。刷新之后window._authority会丢失
import { getAuthority } from '@/utils/authority'; 
import Redirect from 'umi/redirect';
const Authority = getAuthority(); // 获取当前用户权限
const Authorized = RenderAuthorized(Authority);
export default ({ children }) => (
  <Authorized
    authority={children.props.route.authority} // 判断路由配置中的authority是否存在于当前用户权限中，也就是getAuthority()的值
    noMatch={<Redirect to={`/user/login?redirect=${window.location}`} />} // 没有匹配到权限则跳转到登录页
  >
    {children}
  </Authorized>
);

# 代码格式化

暂时解决办法,此方法不能跨编辑器，需要额外安装插件

本人使用格式化快捷键自动修复eslint错误+格式化代码，而不是保存时，这样可以在修改别人代码时不会造成副作用， 实现方式：安装vscode-multi-command插件

// settings.json
{
 "multiCommand.commands": [
    {
      "command": "multiCommand.formatAndfix",
      "sequence": [
        "eslint.executeAutofix",
        // "workbench.action.files.save", 去掉注释可以启用保存时autofix，此设置在全局setting开启也可以
        "editor.action.formatDocument"
      ]
    }
  ]
}

// keybindings.json
{
    "key": "shift+alt+f",
    "command": "multiCommand.formatAndfix",
    "when": "editorHasDocumentFormattingProvider && editorTextFocus && !editorReadonly"
}
